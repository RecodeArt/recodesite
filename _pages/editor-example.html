let numElements = 6; function setup() { //set canvas size createCanvas(800, 800); //define style elements noFill(); strokeWeight(2); //change angles from radians to degrees angleMode(DEGREES); } function draw() { //define style elements background(0); stroke(255); //grid variables let cellWidth = width / numElements; let cellHeight = height / numElements; //grid loop for (let row = 0; row < numElements; row++) { for (let col = 0; col < numElements; col++) { let x = col * cellWidth; let y = row * cellHeight; //grid rectangles rect(x, y, cellWidth, cellHeight); let x1 = random(x, x + cellWidth); let y1 = random(y, y + cellHeight); let x2 = random(x, x + cellWidth); let y2 = random(y, y + cellHeight); let x3 = random(x, x + cellWidth); let y3 = random(y, y + cellHeight); let x4 = random(x, x + cellWidth); let y4 = random(y, y + cellHeight); //lines inside grid line(x1, y1, x2, y2); //circles inside grid circle(x2, y3, random(1, cellWidth)); push(); //randomly rotate squares rotate(random(360)); //squares inside grid rect(x4, y4, random(1, cellWidth / 2)); pop(); } } noLoop(); } //click to redraw shapes in another random order function mousePressed() { redraw(); }
